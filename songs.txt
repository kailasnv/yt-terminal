#!/usr/bin/env python3

import subprocess
import sys

def search_youtube(query):
    try:
        # Use yt-dlp to search and get video URL
        result = subprocess.run(
            ["yt-dlp", f"ytsearch1:{query}", "--get-url"],
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            text=True
        )
        if result.returncode != 0:
            print("Error during search:", result.stderr)
            sys.exit(1)
        return result.stdout.strip()
    except FileNotFoundError:
        print("yt-dlp not found. Please install it.")
        sys.exit(1)

def play_audio(url):
    try:
        # Play using mpv (audio only)
        subprocess.run(["mpv", "--no-video", url])
    except FileNotFoundError:
        print("mpv not found. Please install it.")
        sys.exit(1)

def main():
    if len(sys.argv) < 2:
        print("Usage: play_youtube.py <search term>")
        sys.exit(1)

    query = " ".join(sys.argv[1:])
    print(f"Searching YouTube for: {query}")
    url = search_youtube(query)
    print(f"Playing: {url}")
    play_audio(url)

if __name__ == "__main__":
    main()
